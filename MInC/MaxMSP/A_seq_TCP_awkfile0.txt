function print_notes()
{
	if (notearraysize > 0)
		for (i=0;i<notearraysize;i++)
			print notenum[i]" "notedur[i]
}

BEGIN {
	notearraysize = 0
}

{
	if (count == 0)
	{
		print_notes()

		print $1
	}
	else if (count == 1)
	{
		notearraysize = NF
		for (i=0;i<NF;i++) notenum[i] = $(i+1)
	}
	else if (count == 2)
	{
		for (i=0;i<NF;i++) notedur[i] = $(i+1)
	}
	count = ++count % 3
}

END {
	print_notes()
}
